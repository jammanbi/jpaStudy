-- 회원
CREATE TABLE MEMBER
(
    ID VARCHAR2(30),
    PASSWORD VARCHAR2(50),
    NAME VARCHAR(15),
    GENDER CHAR(1),
    BIRTH_DAY CHAR(8),
    PHONE_NUMBER VARCHAR2(13),
    PROFILE_IMAGE VARCHAR2(100),
    CREATE_DATE DATE
)

ALTER TABLE MEMBER ADD CONSTRAINT MEMBER_PK PRIMARY KEY (ID);

COMMENT ON TABLE MEMBER IS '회원';

COMMENT ON COLUMN MEMBER.ID IS '이메일';
COMMENT ON COLUMN MEMBER.GENDER IS 'W:여, M:남';

COMMIT;
